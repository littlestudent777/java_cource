```java
class Calculator {

    // Метод для суммы двух целых чисел
    public int add(int a, int b) {
        return a + b;
    }

    // Перегруженный метод для суммы трех целых чисел
    public int add(int a, int b, int c) {
        return a + b + c;
    }

  // Возвращают один и тот же тип
}
```
